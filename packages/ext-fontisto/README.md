## Usage

### React Web

```javascript
import Icon from '@svgr-iconkit/fontisto';

export default function App() {
  return <div><Icon name="heart" /></div>
}

```

### React Native

```javascript
import { View } from "react-native";
import Icon from '@svgr-iconkit/fontisto';

export default function App() {
  return <View><Icon name="heart" /></View>
}

```
